内存信息

package main 
import "fmt"
import "runtime"

func main () {
var mem runtime.MemStats
runtime.ReadMemStats(&mem)

// fmt.Println(mem.AccCPU)   // CPU信息 
// fmt.Println(mem.CgoAllocs)   // Cgo调用的Memory分配数量
fmt.Println(mem.GCSys)       // GC回收的内存(M) 
fmt.Println(mem.TotalAlloc)   // 堆内存分配数量(M)
fmt.Println(mem.Mallocs)     // Mallocs的数量
fmt.Print(mem.Frees)        // Frees的数量
}
