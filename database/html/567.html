通道

不带缓冲

package main 
import "fmt"

func main() {
    ch := make(chan int,3)
    ch<-1
    fmt.Printf("通道存放的元素的个数是 %d \n",len(ch))
    fmt.Printf("通道的容量是 %d \n",cap(ch))
    ch<-2
    fmt.Printf("通道存放的元素的个数是 %d \n",len(ch))
    fmt.Printf("通道的容量是 %d",cap(ch))
}
