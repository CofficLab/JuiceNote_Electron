    Legs int
}

    Animal

package main
import "fmt"

type Animal struct {
    Legs int
}

type Dog struct {
    Animal
    Name string
}

func main() {
    dog := Dog{}
    dog.Legs = 4
    dog.Name = "John"

    fmt.Println(dog.Legs)
    fmt.Println(dog.Animal.Legs)
    fmt.Println(dog.Name)
}
