package main
import "errors"
import "fmt"

func myDivide(dividend, divisor float64) (float64, error) {
    if divisor == 0 {
        return 0, errors.New("除数不能为0")
    }

    return dividend / divisor, nil
}

func main() {
    divide, err := myDivide(100, 0)
    if err != nil {
        fmt.Printf("Error: %s", err)
    } else {
        fmt.Printf("100 / 0 = %.2f", divide)
    }
}
