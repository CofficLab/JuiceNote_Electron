String

https://developer.mozilla.org/zh-CN/docs/Web/JavaScript/Reference/Global_Objects/String

定义

var s = "abc";
console.log(typeof s);

属性

长度

按照官方文档的描述，字符串只有一个属性--长度，可以这样获取：

run:

var s = "abc";

console.log(s.length);

方法

concat

str.concat(str2, [, ...strN])

run:

var s = "abc";
var p = "123";
console.log(s.concat(p));

substring

str.substring(indexStart[, indexEnd])

run:

var s = 'abcdef'

// 输出前 2 个字符
console.log(s.substring(0, 2))

常见场景

截取

run:

var s = 'abcdef'

// 输出前 2 个字符
console.log(s.substring(0, 2))

拼接

run:

var s = "abc";
var p = "123";
console.log(s.concat(p));

官方文档推荐使用“+”来拼接：

run:

var a = "老白";
var b = "吃了吗？";

console.log(a + b);

通过 typeof 来获知数据类型：

run:

var s = "龙门飞甲";

console.log(typeof s);

分割

run:

const str = 'a b c d';

console.log(str.split(' '));

填充

run:

const str = "1382345";

console.log(str.padEnd(11, "*"));

转换

  • 转换为数值 <br> run:

    x = "3";
console.log(typeof x);

y = parseInt(x);
console.log(y);
console.log(typeof y);

  • 以某字符串开头 <br> run:

    var a = 'abcdef'
console.log(a.startsWith('ab'))
console.log(a.startsWith(' '))
console.log(a.startsWith(''))

遍历

  • for <br> run:

    var a = 'abcdef'
for (let i = 0; i < a.length; i++) {
console.log(a[i])
}

  • for...in <br> run:

    var a = 'abcdef'
for (let i in a) {
console.log(a[i])
}

  • for...of <br> run:

    var a = 'abcdef'
for (let element of a) {
console.log(element)
}

ASCII 码

  • 获取某个字符的 ASCII 码 <br> run:

    var a = 'abcdef'
console.log(a.charCodeAt(0))
console.log(a.charCodeAt(3))

  • 获取 ASCII 码对应的字符 <br> run:

    console.log(String.fromCharCode(97))
console.log(String.fromCharCode(100))

截取

  • substring <br> run:

    var s = 'abcdef'

// 输出前 2 个字符
console.log(s.substring(0, 2))

  • slice <br> run:

    var s = 'abcdef'

// 输出前 2 个字符
console.log(s.slice(0, 2))
