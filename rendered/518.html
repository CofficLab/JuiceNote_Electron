前缀和后缀

HasPrefix 用于判断字符串是否以某个字符串开头。

package main
import "fmt"
import "strings"

func main() {
    fmt.Println(strings.HasPrefix("老白，吃了吗", "老"))
}

HasSuffix 用于判断字符串是否以某个字符串结尾。

package main
import "fmt"
import "strings"

func main() {
    fmt.Println(strings.HasSuffix("吃了吗？小六", "六"))
}

字符串转数组

package main 
import "fmt"
import "strings"
import "reflect"

func main() {
    a := "莫小贝,佟湘玉,白展堂"
    b := strings.Split(a,",")

    fmt.Println(reflect.TypeOf(b),b)
}

包含关系

Contains 用于判断字符串是否包含某个字符串。

package main
import "fmt"
import "strings"

func main() {
    fmt.Println(strings.Contains("吃了吗？小六", "六"))
}

大小写转换

package main
import "fmt"
import "strings"

func main() {
    fmt.Println(strings.ToLower("AbcDEF"))
    fmt.Println(strings.ToUpper("aBcDEF"))
}

重复

package main
import "fmt"
import "strings"

删除开头的空格

package main

import (
	"fmt"
	"strings"
)

func main() {
	var s = "         Hello"
	s = strings.TrimLeft(s, " ")
	fmt.Print("-"+s+"-")
}

删除两侧的空格

package main

import (
	"fmt"
	"strings"
)

func main() {
	var s = "         Hello       "
	s = strings.Trim(s, " ")
	fmt.Print("-"+s+"-")
}

删除开头的字符串

package main

import (
	"fmt"
	"strings"
)

func main() {
	var s = "¡¡¡Hello, Gophers!!!"
	s = strings.TrimPrefix(s, "¡¡¡Hello, ")
