let password = "123";

if (password == "123") {
  console.log("密码正确");
} else {
  console.log("密码错误");
}

switch...case

官方文档在这里：

<a href="https://developer.mozilla.org/zh-CN/docs/Web/JavaScript/Reference/Statements/switch" target="_blank">https://developer.mozilla.org/zh-CN/docs/Web/JavaScript/Reference/Statements/switch</a>

<div class="run"></div>

let type = "book";

switch (type) {
  case "book":
    console.log("图书");
    break;
  case "chapter":
    console.log("章节");
    break;
  case "page":
    console.log("页面");
    break;
  default:
    console.log("未知");
}
