内部函数

在函数内部，可以声明内部函数。

package main

import "fmt"

func main() {
    var sum func(...int) int // 声明 sum 函数

    sum = func(numbers ...int) int { // 定义 sum 函数
        total := 0
        for _, num := range numbers {
            total += num
        }
        return total
    }

    fmt.Printf("1 + 2 + 3 = %d\n", sum(1, 2, 3))
}
