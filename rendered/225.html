import sqlite3

conn = sqlite3.connect('tmp.db')
curs = conn.cursor()

curs.execute('''
CREATE TABLE IF NOT EXISTS stuff (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    name TEXT
)
''')
curs.execute('''
    INSERT INTO stuff (`name`) VALUES ('佟湘玉')
''')
curs.execute('''
    INSERT INTO stuff (`name`) VALUES ('莫小贝')
''')
conn.commit()
conn.close()

import sqlite3
