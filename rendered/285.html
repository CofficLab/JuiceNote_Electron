那么，字符串的属性有哪些？方法又有哪些？

官方文档里写的很清楚。

属性

长度

按照官方文档的描述，字符串只有一个属性--长度，可以这样获取：

<div class="run"></div>

var s = "abc";

console.log(s.length);

方法

按照官方文档的描述，字符串有很多种方法，用这些方法中的一种或多种组合即可满足我们大部分的计算需要。

随着 JavaScript 版本的不断迭代，有些方法可能会被删除，也可能会增加一些方法。

所以要获取最准确的信息，对官方文档保持关注吧。

以下列举部分场景。

截取

<div class="run"></div>

var s = 'abcdef'

// 输出前 2 个字符
console.log(s.substring(0, 2))

拼接

<div class="run"></div>

var s = "abc";
var p = "123";
console.log(s.concat(p));

官方文档推荐使用“+”来拼接：

<div class="run"></div>

var a = "老白";
var b = "吃了吗？";

console.log(a + b);

通过 typeof 来获知数据类型：

<div class="run"></div>

var s = "龙门飞甲";

console.log(typeof s);

分割

<div class="run"></div>

const str = 'a b c d';

console.log(str.split(' '));

填充

<div class="run"></div>

const str = "1382345";

console.log(str.padEnd(11, "*"));

常见场景

转换

转换为数值

<div class="run"></div>

x = "3";
console.log(typeof x);

y = parseInt(x);
console.log(y);
console.log(typeof y);

以某字符串开头

<div class="run"></div>

var a = 'abcdef'
console.log(a.startsWith('ab'))
console.log(a.startsWith(' '))
console.log(a.startsWith(''))

遍历

for

<div class="run"></div>

var a = 'abcdef'
for (let i = 0; i < a.length; i++) {
  console.log(a[i])
}

for...in

<div class="run"></div>

var a = 'abcdef'
for (let i in a) {
  console.log(a[i])
}

for...of

<div class="run"></div>

var a = 'abcdef'
for (let element of a) {
  console.log(element)
}

ASCII 码

获取某个字符的 ASCII 码

<div class="run"></div>

var a = 'abcdef'
console.log(a.charCodeAt(0))
console.log(a.charCodeAt(3))

获取 ASCII 码对应的字符

<div class="run"></div>

console.log(String.fromCharCode(97))
console.log(String.fromCharCode(100))

var s = 'abcdef'

slice

<div class="run"></div>

var s = 'abcdef'

// 输出前 2 个字符
