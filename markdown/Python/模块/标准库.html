sys

模块 sys 让你能够访问与 Python 解释器紧密相关的变量和函数。

<div class="run"></div>

from sys import argv

print(argv)

<div class="run"></div>

from sys import path

print(path)

<div class="run"></div>

from sys import platform

print(platform)

<div class="run"></div>

from os import environ
