package main
import "fmt"

func nextOf(x int) {
    x = x + 1

    fmt.Printf("nextOf 函数中的x是：%d", x)
}

func main() {
    x :=1
    nextOf(x)

    fmt.Printf("\n")
    fmt.Printf("main 函数中的x是：%d", x)
}

在这里，函数 nextOf 修改了变量 x，但它没变。
